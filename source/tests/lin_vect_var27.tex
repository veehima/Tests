	function getArrayDiff(a, b) {
		var ret = [];
		if (!(Array.isArray(a) && Array.isArray(b))) {
			return ret;
		}
		var i;
		var key;
		
		for (i = a.length - 1; i >= 0; i--) {
			key = a[i];
			if (-1 === b.indexOf(key)) {
				ret.push(key);
			}
		}
		
		return ret;
	}
